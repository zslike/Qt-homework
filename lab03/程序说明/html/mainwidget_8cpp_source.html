177 void mainWidget::handleMarkerClicked()
178 {
179  QLegendMarker* marker = qobject_cast<QLegendMarker*> (sender());
180  Q_ASSERT(marker);
181 
182  switch (marker->type()){
183  case QLegendMarker::LegendMarkerTypeXY:{
184  // 切换选中序列的可见性，
185  marker->series()->setVisible(!marker->series()->isVisible());
186 
187  // 使图注(Legend)的marker也显示出来，因为当隐藏序列series时，
188  // marker也跟着隐藏，从而使用户无法看到marker
189  marker->setVisible(true);
190 
191  // 隐藏序列后虽然显示marker，这里使marker变暗，表明该序列已经被隐藏
192  qreal alpha = 1.0;
193 
194  if (!marker->series()->isVisible()) {
195  alpha = 0.3;
196  }
197 
198  QColor color;
199  QBrush brush = marker->labelBrush();
200  color = brush.color();
201  color.setAlphaF(alpha);
202  brush.setColor(color);
203  marker->setLabelBrush(brush);
204 
205  brush = marker->brush();
206  color = brush.color();
207  color.setAlphaF(alpha);
208  brush.setColor(color);
209  marker->setBrush(brush);
210 
211  QPen pen = marker->pen();
212  color = pen.color();
213  color.setAlphaF(alpha);
214  pen.setColor(color);
215  marker->setPen(pen);
216 
217  break;
218  }
219  default:{
220  qDebug() << "未知的marker类型";
221  break;
222  }
223 
224  }
